    urlgrabber.cpp
    configdialog.cpp
    historycycler.cpp

    klipperpopup.cpp

    editactiondialog.cpp

    clipcommandprocess.cpp
    utils.cpp
    systemclipboard.cpp

    KF6::WindowSystem
    KF6::XmlGui
    Plasma::KWaylandClient
    Wayland::Client
    ${ZLIB_LIBRARY}
)

    target_link_libraries(klipper Qt::GuiPrivate XCB::XCB)

                      VERSION ${PROJECT_VERSION}
                      SOVERSION ${PROJECT_VERSION_MAJOR})
